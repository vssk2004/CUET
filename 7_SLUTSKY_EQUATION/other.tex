\section{Total Change in Demand}


\subsection{Slutsky Identity}

\slide{Slutsky Identity}{
  The total change in demand of good 1, \(\Delta x_1\), is the change in demand due to a
  change in its price, holding income constant:
  \[\Delta x_1 = x_1(p', m) - x_1(p, m)\]

  We have seen above how this change can be broken up into two changes: the substitution eﬀect and the income eﬀect.
  \[\Delta x_1 = \Delta x_1^s + \Delta x_1^n\]
  \[x_1(p',m) - x_1(p,m) = \left[x_1(p',m') - x_1(p,m)\right] + \left[x_1(p',m) - x_1(p',m')\right]\]

  This equation is called the \textbf{Slutsky identity}.
}


\subsection{Total Change in Demand}

\slide{Total Change in Demand}{
  Here, there are four cases:
  \begin{enumerate}
    \item[(1)] Normal Good -- Price Increases
    \item[(2)] Normal Good -- Price Decreases
    \item[(3)] Inferior Good -- Price Increases
    \item[(4)] Inferior Good -- Price Decreases
  \end{enumerate}

  For these, we must analyse the impact of the substitution effect and income effect.
}


\slide{Total Change in Demand -- Normal Good}{
  \begin{enumerate}
    \item[(1)] Normal Good -- Price Increases
    \[\begin{aligned}
    &\text{If price } \uparrow \text{, then SE is } - \text{ve.} \\
    &\text{Price } \uparrow \ \Rightarrow \text{ Real income} \downarrow \text{. For a normal good, IE is } - \text{ve.} \\
    &\text{Total Change in Demand is } - \text{ve.}
    \end{aligned}\]
    \item[(2)] Normal Good -- Price Decreases
    \[\begin{aligned}
    &\text{If price } \downarrow \text{, then SE is } + \text{ve.} \\
    &\text{Price } \downarrow \ \Rightarrow \text{ Real income} \uparrow \text{. For a normal good, IE is } + \text{ve.} \\
    &\text{Total Change in Demand is } + \text{ve.}
    \end{aligned}\]
  \end{enumerate}
}


\slide{Total Change in Demand -- Inferior Good}{
  If the IE is stronger than the SE for an inferior good, we get a Giffen good.
  \begin{enumerate}
    \item[(1)] Inferior Good -- Price Increases
    \[\begin{aligned}
    &\text{If price } \uparrow \text{, then SE is } - \text{ve.} \\
    &\text{Price } \uparrow \ \Rightarrow \text{ Real income} \downarrow \text{. For an inferior good, IE is } + \text{ve.} \\
    &\text{Total Change in Demand is ambiguous. } (\downarrow \text{usually}) \\
    \end{aligned}\]
    \item[(2)] Inferior Good -- Price Decreases
    \[\begin{aligned}
    &\text{If price } \downarrow \text{, then SE is } + \text{ve.} \\
    &\text{Price } \downarrow \ \Rightarrow \text{ Real income} \uparrow \text{. For an inferior good, IE is } - \text{ve.} \\
    &\text{Total Change in Demand is ambiguous. } (\uparrow \text{usually})\\
    \end{aligned}\]
  \end{enumerate}
}
